class DataClass {
}

class Scale extends DataClass {
    min: number = 0;
    max: number = 4;
    step: number = 1;
    labels: Record<number, string> = {};
    images: Record<number, string> = {};
    titles: Record<number, string> = {};
};

class QuestionOption extends DataClass {
    title: string = "";
    value: number = 0;
    label: string = "";
    image: string = ""; 
}

type QuestionOptions = string | Scale | QuestionOption[]

class Question extends DataClass {
    title: string = "";
    description: string = "";
    options: QuestionOptions = [];
    type: string | undefined = undefined;
    questions: Question[] = [];
}

class Quiz extends DataClass {
    title: string = "";
    description: string = "";
    optionsFormats: Record<string, QuestionOptions> = {};
    questions: Question[] = [];
}
